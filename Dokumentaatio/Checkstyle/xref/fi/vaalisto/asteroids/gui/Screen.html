1   package fi.vaalisto.asteroids.gui;
2   
3   import fi.vaalisto.asteroids.logiikka.*;
4   import java.awt.Color;
5   import java.awt.Graphics;
6   import java.awt.Toolkit;
7   import java.util.ArrayList;
8   import java.util.Iterator;
9   import java.util.ListIterator;
10  import java.util.Random;
11  import javax.swing.*;
12  
13  /**
14   * Screen-luokka toteuttaa pelikentän näkyvän osan ja myös aikasilmukan, joka
15   * mahdollistaa pelin reaaliaikaisen toiminnan.
16   *
17   */
18  public class Screen extends JPanel implements Runnable {
19  
20      private static final int FPS = 60; //pelin päivitysnopeus    
21      private static final double MILLIS = 1000 / FPS; //millisekunteja per kuvanpäivitys    
22  
23      public int w;
24      public int h;
25      public Ship ship;
26      public ArrayList<Asteroid> asteroidlist;
27      public ArrayList<Shot> shotlist;
28      public ArrayList<Shot> deadshotlist;
29      public GameKeyListener keylistener;
30      public EventHandler eventhandler;
31  
32      private boolean running = true;
33      private boolean paused = false;
34  
35      /**
36       * Konstruktori asettaa pelikentän leveyden, taustavärin ja alustaa
37       * asteroidit sekä pelaajan aluksen.
38       *
39       * @param w haluttu pelikentän leveys
40       * @param h haluttu pelikentän korkeus
41       */
42      public Screen(int w, int h) {
43          this.w = w;
44          this.h = h;
45          this.asteroidlist = new ArrayList<Asteroid>();
46          this.shotlist = new ArrayList<Shot>();
47          this.deadshotlist = new ArrayList<Shot>();
48          this.setBackground(Color.BLACK);
49          super.setSize(w, h);
50          this.setFocusable(true); //näppäimistökuuntelija ei toimi ilman tätä
51          this.requestFocusInWindow(); //näppäimistökuuntelija ei toimi ilman tätä   
52          this.setDoubleBuffered(true);
53          this.eventhandler = new EventHandler(w, h);
54          initKeyListener();
55      }
56  
57      /**
58       * Alustaa näppäimistökuuntelijan.
59       */
60      public void initKeyListener() {
61          keylistener = new GameKeyListener(this.eventhandler.ship, this);
62          this.addKeyListener(keylistener);
63      }
64  
65      /**
66       * Aluksen piirtämiseen mahdollistava metodi.
67       *
68       * @param g grafiikkaan tarvittava luokka
69       */
70      public void drawShip(Graphics g) {
71          this.eventhandler.ship.draw(g);
72          Toolkit.getDefaultToolkit().sync();
73      }
74  
75      /**
76       * Käydään läpi kaikki asteroidit ja piirretään ne.
77       *
78       * @param g grafiikkaan tarvittava luokka
79       */
80      public void drawAsteroids(Graphics g) {
81          if (!this.eventhandler.asteroidlist.isEmpty()) {
82              for (Asteroid a : this.eventhandler.asteroidlist) {
83                  a.draw(g);
84                  Toolkit.getDefaultToolkit().sync();
85              }
86          }
87      }
88  
89      /**
90       * Käydään läpi kaikki ammukset ja piirretään ne.
91       *
92       * @param g grafiikkaan tarvittava luokka
93       */
94      public void drawShots(Graphics g) {
95          if (!this.eventhandler.shotlist.isEmpty()) {
96              for (Shot s : this.eventhandler.shotlist) {
97                  s.draw(g);
98                  Toolkit.getDefaultToolkit().sync();
99              }
100         }
101     }
102 
103     /**
104      * Vaihdetaan pelin tila tauolle tai sieltä pois.
105      */
106     public void changePause() {
107         if (this.paused) {
108             this.paused = false;
109         } else {
110             this.paused = true;
111         }
112     }
113 
114     /**
115      * Liikutetaan tai poistetaan pelikentällä olevia objekteja.
116      */
117     public void update() {
118         if (!this.paused) {
119             this.eventhandler.ship.move(w, h);
120             this.eventhandler.updateAsteroids();
121             this.eventhandler.updateShots();
122             this.eventhandler.generateShots();
123             this.eventhandler.cleanShots();
124             this.eventhandler.cleanAsteroids();
125         }
126     }
127 
128     @Override
129     public void run() {
130         while (running) {
131             long start = System.currentTimeMillis();
132             update();
133             this.repaint();
134             float deltaTime = (System.currentTimeMillis() - start);
135             int sleepTime = (int) (MILLIS - deltaTime);
136             if (sleepTime > 0) {
137                 try {
138                     Thread.sleep(sleepTime);
139                 } catch (InterruptedException e) {
140                 }
141             }
142         }
143     }
144 
145     @Override
146     protected void paintComponent(Graphics g) {
147         super.paintComponent(g);
148         drawShip(g);
149         drawAsteroids(g);
150         drawShots(g);
151     }
152 }
