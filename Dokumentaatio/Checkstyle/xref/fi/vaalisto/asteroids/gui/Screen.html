1   package fi.vaalisto.asteroids.gui;
2   
3   import fi.vaalisto.asteroids.logiikka.*;
4   import java.awt.Color;
5   import java.awt.Graphics;
6   import java.util.ArrayList;
7   import java.util.ListIterator;
8   import java.util.Random;
9   import javax.swing.*;
10  
11  /**
12   * Screen-luokka toteuttaa pelikentän näkyvän osan ja myös aikasilmukan, joka
13   * mahdollistaa pelin reaaliaikaisen toiminnan.
14   *
15   */
16  public class Screen extends JPanel {
17  
18      private static final int NUMBER_OF_ASTEROIDS = 4; //asteroidien määrä alussa
19      private static final int FPS = 60; //pelin päivitysnopeus
20      private static final double MILLIS = 1000 / FPS; //millisekunteja per kuvanpäivitys
21  
22      public int w;
23      public int h;
24      public Ship ship;
25      public ArrayList<Asteroid> asteroidlist;
26  
27      private boolean running = true;
28      private double delta = 0;
29  
30      /**
31       * Konstruktori asettaa pelikentän leveyden, taustavärin ja alustaa
32       * asteroidit sekä pelaajan aluksen.
33       *
34       * @param w haluttu pelikentän leveys
35       * @param h haluttu pelikentän korkeus
36       */
37      public Screen(int w, int h) {
38          this.w = w;
39          this.h = h;
40          this.asteroidlist = new ArrayList<Asteroid>();
41          this.setBackground(Color.BLUE); //toistaiseksi sininen, jotta näkee objektit paremmin
42          super.setSize(w, h);
43          initShip();
44          initAsteroids();
45      }
46  
47      /**
48       * Asettaa pelaajan aluksen pelikentän keskelle.
49       */
50      public void initShip() {
51          ship = new Ship(w / 2, h / 2);
52      }
53  
54      /**
55       * Alustaa halutun määrän uusia asteroideja ja satunnaisgeneroi niiden
56       * sijainnit.
57       */
58      public void initAsteroids() {
59          for (int i = 0; i < NUMBER_OF_ASTEROIDS; i++) {
60              int randomX = (int) (Math.random() * w);
61              int randomY = (int) (Math.random() * h);
62              asteroidlist.add(new Asteroid(randomX, randomY));
63          }
64      }
65  
66      /**
67       * Aluksen piirtämiseen mahdollistava metodi.
68       *
69       * @param g grafiikkaan tarvittava luokka
70       */
71      public void drawShip(Graphics g) {
72          this.ship.draw(g);
73      }
74  
75      /**
76       * Käydään läpi kaikki asteroidit ja piirretään ne.
77       *
78       * @param g grafiikkaan tarvittava luokka
79       */
80      public void drawAsteroids(Graphics g) {
81          if (!asteroidlist.isEmpty()) {
82              for (Asteroid a : asteroidlist) {
83                  a.draw(g);
84              }
85          }
86      }
87  
88      /**
89       * Liikutetaan jokaista asteroidia.
90       */
91      public void updateAsteroids() {
92          if (!asteroidlist.isEmpty()) {
93              for (Asteroid a : asteroidlist) {
94                  a.move(w, h);
95              }
96          }
97      }
98  
99      /**
100      * Liikutetaan pelikentän kaikkia tarvittavia olioita
101      */
102     public void update() {
103         ship.move(w, h);
104         updateAsteroids();
105     }
106 
107     /**
108      * Pelin reaaliaikaisen toiminnan mahdollistava metodi.
109      */
110     public void run() {
111         long lastTime = System.currentTimeMillis();
112         while (running) {
113             long now = System.currentTimeMillis();
114             delta += (now - lastTime) / MILLIS;
115             lastTime = now;
116             while (delta >= 1) {
117                 update();
118                 this.repaint();
119             }
120         }
121     }
122 
123     /**
124      * JPanelilta peritty metodi, joka piirtää kaikki graafiset komponentit
125      *
126      * @param g grafiikkaan tarvittava luokka
127      */
128     @Override
129     protected void paintComponent(Graphics g) {
130         super.paintComponent(g);
131         drawShip(g);
132         drawAsteroids(g);
133     }
134 }
