3   import java.awt.*;
4   import java.io.IOException;

7   /**
8    * Ship-luokka toteuttaa pelaajan ohjaaman aluksen.
9    *
10   */
11  public class Ship extends GameObj {
12  
13      private static final double TURN_RATE = 0.000001; //aluksen kääntymisnopeus
14      private static final double SPEED_LIMIT = 0.00003; //aluksen maksiminopeus
15      private static final double ACCELERATION = 0.000000000005; //aluksen kiihtyvyys
16  
17      boolean accelerating; // totuusarvo siitä kiihdyttääkö alus. Alus on ainoa objekti, joka voi kiihdyttää.
18      boolean turningLeft, turningRight; // totuusarvo siitä kääntyykö alus. Alus on ainoa objekti, joka voi kääntyä.
19  
20      /**
21       * Ship-luokan konstruktori. Nämä jätetään muuttujiksi ja aluksen speksejä
22       * voidaan muuttaa pelin toteuttavassa luokassa sen sijaan, että
23       * kovakoodattaisiin tähän luokkaan.
24       *
25       * @param x aluksen aloituspaikan x-koordinaatti
26       * @param y aluksen aloituspaikan y-koordinaatti
27       *
28       */
29      public Ship(int x, int y) {
30          this.x = x;
31          this.y = y;
32          this.angle = 0;
33          this.xVelocity = 0;
34          this.yVelocity = 0;
35          this.accelerating = true; //väliaikainen arvo testausta varten
36          this.turningLeft = true; //väliaikainen arvo testausta varten
37          this.turningRight = false;
38          this.img = null;
39          try {
40              this.img = ImageIO.read(getClass().getClassLoader().getResourceAsStream("ship.png"));
41  
42          } catch (IOException e) {
43              System.out.println("Ship picture missing!");
44          }
45  
46      }
47  
48      public boolean isAccelerating() {
49          return accelerating;
50      }
51  
52      public void setAccelerating(boolean accelerating) {
53          this.accelerating = accelerating;
54      }
55  
56      public boolean isTurningLeft() {
57          return turningLeft;
58      }
59  
60      public void setTurningLeft(boolean turningLeft) {
61          this.turningLeft = turningLeft;
62      }
63  
64      public boolean isTurningRight() {
65          return turningRight;
66      }
67  
68      public void setTurningRight(boolean turningRight) {
69          this.turningRight = turningRight;
70      }
71  
72      /**
73       * Lasketaan nopeuskomponenttien summavektorin pituus, jotta voidaan pitää
74       * se maksiminopeuden rajoissa.
75       *
76       * @param xComponent aluksen nopeuden x-komponentti
77       * @param yComponent aluksen nopeuden y-komponentti
78       * @return summavektorin pituus
79       */
80      private double sumSpeedVector(double xComponent, double yComponent) {
81          return Math.sqrt((xComponent * xComponent) + (yComponent * yComponent));
82      }
83  
84      /**
85       * Alukselle laajennettu liikkuminen. Alus on aina olio, joka voi kääntyä ja
86       * kiihtyä. *
87       *
88       * @param screenWidth pelikentän leveys
89       * @param screenHeight peikentän korkeus
90       */
91      @Override
92      public void move(int screenWidth, int screenHeight) {
93          super.move(screenWidth, screenHeight);
94          if (turningLeft) {
95              angle -= TURN_RATE;
96          }
97          if (turningRight) {
98              angle += TURN_RATE;
99          }
100         angleCheck();
101         if (accelerating) {
102             double deltaXVel = ACCELERATION * Math.sin(angle);
103             double deltaYVel = ACCELERATION * Math.cos(angle);
104             double newXVel = xVelocity + deltaXVel;
105             double newYVel = yVelocity + deltaYVel;
106             double newSpeed = sumSpeedVector(newXVel, newYVel);
107             if (newSpeed < SPEED_LIMIT) {
108                 xVelocity -= deltaXVel;
109                 yVelocity += deltaYVel;
110             }
111 
112         }
113     }
