3   import java.awt.*;
4   import java.io.IOException;
5   import javax.imageio.ImageIO;

7   /**
8    * Ship-luokka toteuttaa pelaajan ohjaaman aluksen.
9    *
10   */
11  public class Ship extends GameObj {
12  
13      private static final double TURN_RATE = 0.1; //aluksen kääntymisnopeus
14      private static final double SPEED_LIMIT = 1.5; //aluksen maksiminopeus
15      private static final double ACCELERATION = 0.1; //aluksen kiihtyvyys
16      private static final double VELOCITY_DECAY = 0.98; //aluksen "kitka"
17      private static final int SHOOT_DELAY_RESET = 30;
18      private static final String IMG_NAME = "ship.png";
19  
20      public boolean accelerating; // totuusarvo siitä kiihdyttääkö alus. Alus on ainoa objekti, joka voi kiihdyttää.
21      public boolean turningLeft, turningRight; // totuusarvo siitä kääntyykö alus. Alus on ainoa objekti, joka voi kääntyä.
22      public boolean shooting;
23      public int shootDelay;
24  
25      /**
26       * Ship-luokan konstruktori. Nämä jätetään muuttujiksi ja aluksen speksejä
27       * voidaan muuttaa pelin toteuttavassa luokassa sen sijaan, että
28       * kovakoodattaisiin tähän luokkaan.
29       *
30       * @param x aluksen aloituspaikan x-koordinaatti
31       * @param y aluksen aloituspaikan y-koordinaatti
32       *
33       */
34      public Ship(int x, int y) {
35          this.x = x;
36          this.y = y;
37          this.angle = 0;
38          this.xVelocity = 0;
39          this.yVelocity = 0;
40          this.accelerating = false; //väliaikainen arvo testausta varten
41          this.turningLeft = false; //väliaikainen arvo testausta varten
42          this.turningRight = false;
43          this.shooting = false;
44          this.shootDelay = 0;
45          this.img = null;
46          try {
47              this.img = ImageIO.read(getClass().getClassLoader().getResourceAsStream(IMG_NAME));
48  
49          } catch (IOException e) {
50              System.out.println("Ship picture missing!");
51          }
52  
53      }
54  
55      public boolean isAccelerating() {
56          return accelerating;
57      }
58  
59      public void setAccelerating(boolean accelerating) {
60          this.accelerating = accelerating;
61      }
62  
63      public boolean isTurningLeft() {
64          return turningLeft;
65      }
66  
67      public void setTurningLeft(boolean turningLeft) {
68          this.turningLeft = turningLeft;
69      }
70  
71      public boolean isTurningRight() {
72          return turningRight;
73      }
74  
75      public void setTurningRight(boolean turningRight) {
76          this.turningRight = turningRight;
77      }
78  
79      public boolean isShooting() {
80          return shooting;
81      }
82  
83      public void setShooting(boolean shooting) {
84          this.shooting = shooting;
85      }
86  
87      public int getShootDelay() {
88          return shootDelay;
89      }
90  
91      /**
92       * Aluksen ampuessa tehdään uusi Shot-olio.
93       *
94       * @return Shot-olio, joka näkyy pelissä ammusena.
95       */
96      public Shot shoots() {
97          this.shootDelay = SHOOT_DELAY_RESET;
98          return new Shot(this.x, this.y, this.angle, this.xVelocity, this.yVelocity);
99      }
100 
101     /**
102      * Aluksen liikkumismetodin kääntymismetodi.
103      */
104     private void turn() {
105         if (turningLeft) {
106             angle -= TURN_RATE;
107         }
108         if (turningRight) {
109             angle += TURN_RATE;
110         }
111     }
112     
113     /**
114      * Aluksen liikkumismetodin kiihdyttämismetodi.
115      */
116 
117     private void accelerate() {
118         if (accelerating) {
119             xVelocity -= ACCELERATION * Math.sin(angle);
120             yVelocity += ACCELERATION * Math.cos(angle);
121         }
122     }
123     
124     /**
125      * Aluksen liikkumismetodin nopeutta vähentävä metodi.
126      */
127 
128     private void decaySpeed() {
129         xVelocity *= VELOCITY_DECAY;
130         yVelocity *= VELOCITY_DECAY;
131     }
132     
133     /**
134      * Vähentää laskuria, jolla määritellään aluksen tulinopeus.
135      */
136     private void decreaseShotDelay() {
137         if (shootDelay > 0) {
138             shootDelay--;
139         }
140     }
141 
142     /**
143      * Alukselle laajennettu liikkuminen. Alus on ainoa olio, joka voi kääntyä
144      * ja kiihtyä. Metodi on jaettu useampaan metodiin.
145      *
146      * @param screenWidth pelikentän leveys
147      * @param screenHeight peikentän korkeus
148      */
149     @Override
150     public void move(int screenWidth, int screenHeight) {
151         super.move(screenWidth, screenHeight);
152         turn();
153         angleCheck();
154         accelerate();
155         decaySpeed();
156         decreaseShotDelay();
157     }
158 
159 }
