4   import java.io.IOException;

7   /**

9    * @author Ville Vaalisto

11  public class Ship extends GameObj {
12  
13      boolean accelerating; // totuusarvo siitä kiihdyttääkö alus. Alus on ainoa objekti, joka voi kiihdyttää.
14      boolean turningLeft, turningRight; // totuusarvo siitä kääntyykö alus. Alus on ainoa objekti, joka voi kääntyä.
15  
16      /**
17       * Ship-luokan konstruktori. Nämä jätetään muuttujiksi ja aluksen speksejä
18       * voidaan muuttaa pelin toteuttavassa luokassa sen sijaan, että
19       * kovakoodattaisiin tähän luokkaan.
20       *
21       * @param x aluksen aloituspaikan x-koordinaatti
22       * @param y aluksen aloituspaikan y-koordinaatti
23       *
24       */
25      public Ship(int x, int y) {
26          this.x = x;
27          this.y = y;
28          this.angle = 0;
29          this.xVelocity = 0;
30          this.yVelocity = 0;
31          this.acceleration = 0.35; //mielivaltainen arvo tällä hetkellä.
32          this.accelerating = false;
33          this.turningLeft = false;
34          this.turningRight = false;
35          this.img = null;
36          try {
37              this.img = ImageIO.read(getClass().getClassLoader().getResourceAsStream("ship.png"));
38          } catch (IOException e) {
39              System.out.println("Ship picture missing!");
40          }
41  
42      }
43  
44  }
