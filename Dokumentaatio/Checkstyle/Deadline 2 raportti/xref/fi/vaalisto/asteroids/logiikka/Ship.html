9   public class Ship extends GameObj {
10  
11      boolean accelerating; // totuusarvo siitä kiihdyttääkö alus. Alus on ainoa objekti, joka voi kiihdyttää.
12      boolean turningLeft, turningRight; // totuusarvo siitä kääntyykö alus. Alus on ainoa objekti, joka voi kääntyä.
13  
14      /**
15       * Ship-luokan konstruktori. Nämä jätetään muuttujiksi ja aluksen speksejä
16       * voidaan muuttaa pelin toteuttavassa luokassa sen sijaan, että
17       * kovakoodattaisiin tähän luokkaan.
18       *
19       * @param x aluksen aloituspaikan x-koordinaatti
20       * @param y aluksen aloituspaikan y-koordinaatti
21       *
22       */
23      public Ship(int x, int y) {
24          this.x = x;
25          this.y = y;
26          this.angle = 0;
27          this.xVelocity = 0;
28          this.yVelocity = 0;
29          this.acceleration = 0.35; //mielivaltainen arvo tällä hetkellä.
30          this.accelerating = false;
31          this.turningLeft = false;
32          this.turningRight = false;
33  
34      }

36  }
